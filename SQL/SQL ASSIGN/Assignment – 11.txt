2) Write a query that produces the names and ratings of all customers who have 
above-average orders. 


SELECT c.cname, c.rating
FROM customers c
JOIN (
  SELECT cnum, COUNT(*) as order_count
  FROM orders
  GROUP BY cnum
) o ON c.cnum = o.cnum
WHERE o.order_count = (
  SELECT AVG(order_count)
  FROM (
    SELECT cnum, COUNT(*) as order_count
    FROM orders
    GROUP BY cnum
  ) subquery
);
